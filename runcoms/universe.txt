##==
#
#   project basic settings
#
##==
cmake_minimum_required(VERSION 3.3)
project(universe)

##==
#
#   compiler options
#
##==
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11 -stdlib=libc++")
set(CMAKE_INCLUDE_CURRENT_DIR true)

set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/build/lib)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/build/lib)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/build/bin)

##==
#
#   include and lib locations
#
##==
include_directories(include)
link_directories(lib)

##==
#
#   C++11 source files
#
##==
set(SOURCE_FILES src/cpp11/41-concurrency-future.cpp)
set(SOURCE_FILES src/cpp11/36-string.cpp)
set(SOURCE_FILES src/cpp11/11-operations-lambda.cpp)
set(SOURCE_FILES src/cpp11/14-namespace-unnamed.cpp)

##==
#
#   base C
#
##==
set(SOURCE_FILES src/base/daemon.c)
set(SOURCE_FILES src/base/base_test.c)

include_directories(src/base)
link_directories(src/base)

##==
#
#   universe main entry
#
##==
set(SOURCE_FILES src/main/main.cpp)

include_directories(libraries/boost/include)
link_directories(libraries/boost/lib)

##==
#
#   asio examples
#
##==
add_executable(asio_echo_server src/asio/echo/blocking_tcp_echo_server.cpp)
target_link_libraries(asio_echo_server boost_system)

add_executable(asio_echo_client src/asio/echo/blocking_tcp_echo_client.cpp)
target_link_libraries(asio_echo_client boost_system)


##==
#
#   template - hello
#
##==
set(SOURCE_FILES snp/template.hello.cpp)


##===----------------------------------------------------------------------===##

set(SOURCE_FILES src/cpp11/14-namespace-unnamed.cpp)

add_definitions(-DHAVE_CXX_STDHEADERS)
add_executable(universe ${SOURCE_FILES})
target_link_libraries (universe boost_system)
